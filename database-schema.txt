Table Questions{
  id int [pk, increment]
  title varchar
  content text
  views int
  userId int
  topicId int
}

Table FollowsBONUS{
  id int [pk, increment]
  questionId int
  userId int
}

Table Answers{
  id int [pk, increment]
  content text
  userId int
  questionId int
}

Table Comments{
  id int [pk, increment]
  content text
  answerId int
  userId int
}

Table Users{
  id int [pk, increment]
  firstName varchar
  lastName varchar
  email varchar
  username varchar
  password password
}

Table Topic{
  id int [pk, increment]
  name varchar
}

Table SubCommentsBONUS{
  id int [pk, increment]
  content text
  upVotes int
  donwVotes int
  userId int
  commentId int
}
