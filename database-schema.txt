Table Questions{
  id int [pk, increment]
  title varchar
  content text
  imageUrl varchar
  follows int
  upvotes int
  userId int
  topicId int
}

Table Follows{
  id int [pk, increment]
  questionId int
  userId int
}

Table Answers{
  id int [pk, increment]
  content text
  userId int
  questionId int
}

Table Comments{
  id int [pk, increment]
  content text
  answerId int
  userId int
  subCommentId int
}

Table Users{
  id int [pk, increment]
  firstName varchar
  lastName varchar
  email varchar
  username varchar
  password password
}

Table Topic{
  id int [pk, increment]
  name varchar
}

Table SubComments{
  id int [pk, increment]
  content text
  userId int
  commentId int
}

Table Upvotes{
  id int [pk, increment]
  questionId int
  userId int
}


Ref: "Answers"."questionId" > "Questions"."id"

Ref: "Answers"."userId" > "Users"."id"

Ref: "Comments"."answerId" > "Answers"."id"

Ref: "Topic"."id" < "Questions"."topicId"

Ref: "Questions"."userId" > "Users"."id"

Ref: "Users"."id" < "Comments"."userId"

Ref: "Follows"."questionId" < "Questions"."follows"

Ref: "Follows"."userId" < "Users"."id"

Ref: "Users"."id" < "SubComments"."userId"

Ref: "Comments"."subCommentId" < "SubComments"."commentId"

Ref: "Upvotes"."questionId" < "Questions"."upvotes"

Ref: "Users"."id" < "Upvotes"."userId"
